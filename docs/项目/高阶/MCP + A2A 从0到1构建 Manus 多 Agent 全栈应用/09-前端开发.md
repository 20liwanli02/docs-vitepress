## 三、SSE 流式响应处理

### 3.1 SSE Hook

```tsx
// src/hooks/use-sse.ts

import { useCallback, useRef } from "react"
import { Event } from "@/types"

export function useSSE() {
  const eventSourceRef = useRef<EventSource | null>(null)
  
  const connect = useCallback((
    url: string,
    onMessage: (event: Event) => void,
    onError?: (error: Error) => void
  ) => {
    // 关闭旧连接
    if (eventSourceRef.current) {
      eventSourceRef.current.close()
    }
    
    // 创建新连接
    const eventSource = new EventSource(url)
    eventSourceRef.current = eventSource
    
    // 监听消息
    eventSource.onmessage = (e) => {
      try {
        const event = JSON.parse(e.data) as Event
        onMessage(event)
        
        // 如果是结束事件，关闭连接
        if (event.type === "done") {
          eventSource.close()
        }
      } catch (err) {
        console.error("解析事件失败:", err)
      }
    }
    
    // 监听错误
    eventSource.onerror = (e) => {
      console.error("SSE 连接错误:", e)
      onError?.(new Error("SSE 连接错误"))
      eventSource.close()
    }
    
    return () => {
      eventSource.close()
    }
  }, [])
  
  const disconnect = useCallback(() => {
    if (eventSourceRef.current) {
      eventSourceRef.current.close()
      eventSourceRef.current = null
    }
  }, [])
  
  return { connect, disconnect }
}
```

### 3.2 聊天 Hook

```tsx
// src/hooks/use-chat.ts

import { useState, useCallback } from "react"
import { useSSE } from "./use-sse"
import { Event, Message } from "@/types"

export function useChat(sessionId: string) {
  const [messages, setMessages] = useState<Message[]>([])
  const [isLoading, setIsLoading] = useState(false)
  const { connect, disconnect } = useSSE()
  
  const sendMessage = useCallback(async (content: string) => {
    // 添加用户消息
    const userMessage: Message = {
      id: Date.now().toString(),
      role: "user",
      content,
      createdAt: new Date().toISOString()
    }
    setMessages(prev => [...prev, userMessage])
    setIsLoading(true)
    
    // 发送请求
    const response = await fetch(`/api/sessions/${sessionId}/chat`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ message: content })
    })
    
    // 处理 SSE 响应
    const reader = response.body?.getReader()
    const decoder = new TextDecoder()
    
    let currentMessage: Message = {
      id: "",
      role: "assistant",
      content: "",
      createdAt: new Date().toISOString()
    }
    
    while (reader) {
      const { done, value } = await reader.read()
      if (done) break
      
      const chunk = decoder.decode(value)
      const lines = chunk.split("\n")
      
      for (const line of lines) {
        if (line.startsWith("data: ")) {
          const data = line.slice(6)
          const event = JSON.parse(data) as Event
          
          // 根据事件类型更新消息
          switch (event.type) {
            case "message":
              currentMessage.content += event.message
              setMessages(prev => {
                const filtered = prev.filter(m => m.id !== currentMessage.id)
                return [...filtered, currentMessage]
              })
              break
              
            case "tool":
              // 添加工具调用信息
              currentMessage.toolCalls = currentMessage.toolCalls || []
              currentMessage.toolCalls.push(event)
              setMessages(prev => [...prev])
              break
              
            case "done":
              setIsLoading(false)
              break
          }
        }
      }
    }
  }, [sessionId])
  
  return {
    messages,
    isLoading,
    sendMessage
  }
}
```

```tsx
import { ToolEvent } from "@/types"
import { Wrench, CheckCircle, XCircle, Loader2 } from "lucide-react"

interface ToolCallDisplayProps {
  toolCall: ToolEvent
}

export function ToolCallDisplay({ toolCall }: ToolCallDisplayProps) {

  const isSuccess = toolCall.status === "called" && toolCall.functionResult?.success
  
  return (
    <div className="bg-gray-50 rounded-lg p-3 text-sm">
```

## 五、VNC 预览组件

### 5.1 VNC 组件

```tsx
// src/components/vnc-viewer.tsx

"use client"

import { useEffect, useRef, useState } from "react"
import RFB from "@novnc/novnc/core/rfb"

interface VNCViewerProps {
  url: string
  className?: string
}

export function VNCViewer({ url, className }: VNCViewerProps) {
  const containerRef = useRef<HTMLDivElement>(null)
  const rfbRef = useRef<RFB | null>(null)
  const [connected, setConnected] = useState(false)
  const [error, setError] = useState<string | null>(null)
  
  useEffect(() => {
    if (!containerRef.current || !url) return
    
    // 创建 VNC 连接
    try {
      rfbRef.current = new RFB(containerRef.current, url, {
        credentials: { password: "" }
      })
      
      rfbRef.current.scaleViewport = true
      rfbRef.current.resizeSession = true
      
      rfbRef.current.addEventListener("connect", () => {
        setConnected(true)
        setError(null)
      })
      
      rfbRef.current.addEventListener("disconnect", () => {
        setConnected(false)
      })
      
      rfbRef.current.addEventListener("securityfailure", (e) => {
        setError("安全验证失败")
      })
    } catch (err) {
      setError("连接失败")
    }
    
    return () => {
      if (rfbRef.current) {
        rfbRef.current.disconnect()
        rfbRef.current = null
      }
    }
  }, [url])
  
  return (
    <div className={className}>
      {/* 状态栏 */}
      <div className="flex items-center justify-between p-2 bg-gray-100">
        <span className="text-sm">
          {connected ? (
            <span className="text-green-500">● 已连接</span>
          ) : (
            <span className="text-gray-500">○ 未连接</span>
          )}
        </span>
      </div>
      
      {/* VNC 画面 */}
      <div 
        ref={containerRef}
        className="w-full h-full bg-black"
      />
      
      {/* 错误提示 */}
      {error && (
        <div className="absolute inset-0 flex items-center justify-center bg-black/50">
          <p className="text-white">{error}</p>
        </div>
      )}
    </div>
  )
}
```

## 六、文件预览组件

### 6.1 文件列表组件

```tsx
// src/components/file-list.tsx

import { File, Folder, Code, FileText, Image } from "lucide-react"

interface FileItem {
  name: string
  path: string
  type: "file" | "directory"
  size?: number
  mimeType?: string
}

interface FileListProps {
  files: FileItem[]
  onSelect: (file: FileItem) => void
}

export function FileList({ files, onSelect }: FileListProps) {
  const getFileIcon = (file: FileItem) => {
    if (file.type === "directory") return <Folder className="w-5 h-5 text-yellow-500" />
    
    const ext = file.name.split(".").pop()?.toLowerCase()
    switch (ext) {
      case "js":
      case "ts":
      case "py":
      case "java":
        return <Code className="w-5 h-5 text-blue-500" />
      case "md":
      case "txt":
        return <FileText className="w-5 h-5 text-gray-500" />
      case "png":
      case "jpg":
      case "gif":
        return <Image className="w-5 h-5 text-green-500" />
      default:
        return <File className="w-5 h-5 text-gray-400" />
    }
  }
  
  return (
    <div className="divide-y">
      {files.map((file, index) => (
        <div
          key={index}
          className="flex items-center gap-3 p-3 hover:bg-gray-50 cursor-pointer"
          onClick={() => onSelect(file)}
        >
          {getFileIcon(file)}
          <div className="flex-1 min-w-0">
            <p className="truncate font-medium">{file.name}</p>
            <p className="text-xs text-gray-500">{file.path}</p>
          </div>
          {file.size && (
            <span className="text-xs text-gray-400">
              {formatFileSize(file.size)}
            </span>
          )}
        </div>
      ))}
    </div>
  )
}

function formatFileSize(bytes: number): string {
  if (bytes < 1024) return `${bytes} B`
  if (bytes < 1024 * 1024) return `${(bytes / 1024).toFixed(1)} KB`
  return `${(bytes / (1024 * 1024)).toFixed(1)} MB`
}
```

## 七、API 调用封装

### 7.1 API 客户端

```tsx
// src/lib/api.ts

import { Session, Message, File } from "@/types"

const API_BASE = "/api"

export const api = {
  // 会话相关
  sessions: {
    list: async (): Promise<Session[]> => {
      const res = await fetch(`${API_BASE}/sessions`)
      return res.json()
    },
    
    get: async (id: string): Promise<Session> => {
      const res = await fetch(`${API_BASE}/sessions/${id}`)
      return res.json()
    },
    
    create: async (): Promise<Session> => {
      const res = await fetch(`${API_BASE}/sessions`, { method: "POST" })
      return res.json()
    },
    
    delete: async (id: string): Promise<void> => {
      await fetch(`${API_BASE}/sessions/${id}`, { method: "DELETE" })
    }
  },
  
  // 聊天相关
  chat: {
    send: async (sessionId: string, message: string): Promise<Response> => {
      return fetch(`${API_BASE}/sessions/${sessionId}/chat`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ message })
      })
    }
  },
  
  // 文件相关
  files: {
    list: async (sessionId: string): Promise<File[]> => {
      const res = await fetch(`${API_BASE}/sessions/${sessionId}/files`)
      return res.json()
    },
    
    get: async (sessionId: string, path: string): Promise<Blob> => {
      const res = await fetch(`${API_BASE}/sessions/${sessionId}/files/${encodeURIComponent(path)}`)
      return res.blob()
    }
  }
}
```
